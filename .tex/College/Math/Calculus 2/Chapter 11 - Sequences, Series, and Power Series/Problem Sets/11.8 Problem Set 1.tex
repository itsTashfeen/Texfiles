\section*{Concept Checklist}
This problem set is designed to test your understanding of the following concepts related to Power Series. The goal is to ensure you have mastered the full range of techniques required for this topic.

\begin{itemize}
    \item \textbf{Core Technique: Finding the Radius of Convergence}
    \begin{itemize}
        \item Applying the Ratio Test to a power series of the form $\sum c_n (x-a)^n$.
        \item Applying the Root Test, especially for series where $c_n$ involves an $n$-th power.
    \end{itemize}
    
    \item \textbf{Endpoint Analysis: Determining the Interval of Convergence}
    \begin{itemize}
        \item \textbf{p-Series Test:} Recognizing and applying the test for series of the form $\sum \frac{1}{n^p}$ at an endpoint.
        \item \textbf{Alternating Series Test (AST):} Applying the test for series with a $(-1)^n$ term at an endpoint.
        \item \textbf{Divergence Test:} Identifying series whose terms do not approach zero at the endpoints.
        \item \textbf{Limit Comparison Test (LCT):} Using the test for more complex endpoint series by comparing them to known series like $\sum \frac{1}{n}$.
    \end{itemize}
    
    \item \textbf{Categorization of Power Series}
    \begin{itemize}
        \item Series centered at $a=0$.
        \item Series centered at $a \neq 0$, including those requiring algebraic manipulation (e.g., rewriting $(kx-c)^n$ as $k^n(x-c/k)^n$).
    \end{itemize}

    \item \textbf{Special Cases for the Radius of Convergence}
    \begin{itemize}
        \item Series that converge only at their center ($R=0$).
        \item Series that converge for all real numbers ($R=\infty$).
    \end{itemize}

    \item \textbf{Advanced Concepts}
    \begin{itemize}
        \item Series with more complex terms, such as $x^{2n}$, $x^{n^2}$, or factorials in both numerator and denominator.
        \item Finding new series through term-by-term differentiation and integration of a known series.
    \end{itemize}
\end{itemize}

\section*{Solutions}

\subsection*{Solutions to Standard Series (Centered at a=0)}

\textbf{1. $\displaystyle \sum_{n=1}^{\infty} \frac{x^n}{n^3}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{x^{n+1}}{(n+1)^3} \cdot \frac{n^3}{x^n} \right| = |x| \lim_{n \to \infty} \left(\frac{n}{n+1}\right)^3 = |x|$.
    \item For convergence, $|x|<1$. Thus, $R=1$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=1$: $\sum \frac{1}{n^3}$. Converges (p-series, $p=3>1$).
            \item $x=-1$: $\sum \frac{(-1)^n}{n^3}$. Converges (absolutely).
        \end{itemize}
    \item \textbf{Answer:} $R=1, I=[-1,1]$.
\end{itemize}

\textbf{2. $\displaystyle \sum_{n=1}^{\infty} \frac{(-1)^n x^n}{n \cdot 5^n}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{x^{n+1}}{(n+1)5^{n+1}} \cdot \frac{n \cdot 5^n}{x^n} \right| = \frac{|x|}{5} \lim_{n \to \infty} \frac{n}{n+1} = \frac{|x|}{5}$.
    \item For convergence, $\frac{|x|}{5}<1 \implies |x|<5$. Thus, $R=5$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=5$: $\sum \frac{(-1)^n 5^n}{n \cdot 5^n} = \sum \frac{(-1)^n}{n}$. Converges (AST).
            \item $x=-5$: $\sum \frac{(-1)^n (-5)^n}{n \cdot 5^n} = \sum \frac{5^n}{n \cdot 5^n} = \sum \frac{1}{n}$. Diverges (harmonic).
        \end{itemize}
    \item \textbf{Answer:} $R=5, I=(-5,5]$.
\end{itemize}

\textbf{3. $\displaystyle \sum_{n=1}^{\infty} \frac{n^2 x^n}{3^n}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(n+1)^2 x^{n+1}}{3^{n+1}} \cdot \frac{3^n}{n^2 x^n} \right| = \frac{|x|}{3} \lim_{n \to \infty} \left(\frac{n+1}{n}\right)^2 = \frac{|x|}{3}$.
    \item For convergence, $|x|<3$. Thus, $R=3$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=3$: $\sum \frac{n^2 3^n}{3^n} = \sum n^2$. Diverges (Divergence Test).
            \item $x=-3$: $\sum \frac{n^2 (-3)^n}{3^n} = \sum (-1)^n n^2$. Diverges (Divergence Test).
        \end{itemize}
    \item \textbf{Answer:} $R=3, I=(-3,3)$.
\end{itemize}

\textbf{4. $\displaystyle \sum_{n=1}^{\infty} n^n x^n$}
\begin{itemize}
    \item \textbf{Root Test:} $L = \lim_{n \to \infty} \sqrt[n]{|n^n x^n|} = \lim_{n \to \infty} |nx| = \infty$ for any $x \neq 0$.
    \item The series diverges for all $x \neq 0$. It only converges at its center, $x=0$.
    \item \textbf{Answer:} $R=0, I=\{0\}$.
\end{itemize}

\textbf{5. $\displaystyle \sum_{n=2}^{\infty} \frac{x^n}{\ln n}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{x^{n+1}}{\ln(n+1)} \cdot \frac{\ln n}{x^n} \right| = |x| \lim_{n \to \infty} \frac{\ln n}{\ln(n+1)} = |x| \cdot 1 = |x|$ (using L'Hopital's Rule).
    \item For convergence, $|x|<1$. Thus, $R=1$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=1$: $\sum \frac{1}{\ln n}$. Diverges (Direct comparison with $\sum \frac{1}{n}$, since $\ln n < n$ for $n \ge 1$).
            \item $x=-1$: $\sum \frac{(-1)^n}{\ln n}$. Converges (AST).
        \end{itemize}
    \item \textbf{Answer:} $R=1, I=[-1,1)$.
\end{itemize}

\textbf{6. $\displaystyle \sum_{n=1}^{\infty} \frac{10^n x^n}{n^3}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{10^{n+1} x^{n+1}}{(n+1)^3} \cdot \frac{n^3}{10^n x^n} \right| = 10|x| \lim_{n \to \infty} (\frac{n}{n+1})^3 = 10|x|$.
    \item For convergence, $10|x|<1 \implies |x|<1/10$. Thus, $R=1/10$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=1/10$: $\sum \frac{10^n (1/10)^n}{n^3} = \sum \frac{1}{n^3}$. Converges (p-series, $p=3>1$).
            \item $x=-1/10$: $\sum \frac{10^n (-1/10)^n}{n^3} = \sum \frac{(-1)^n}{n^3}$. Converges (absolutely).
        \end{itemize}
    \item \textbf{Answer:} $R=1/10, I=[-1/10, 1/10]$.
\end{itemize}

\textbf{7. $\displaystyle \sum_{n=1}^{\infty} \frac{(-1)^n x^{2n}}{9^n}$}
\begin{itemize}
    \item This is a geometric series in disguise. Let $u = -x^2/9$. The series is $\sum_{n=1}^\infty u^n$. It converges if $|u|<1$.
    \item $|-x^2/9|<1 \implies x^2/9 < 1 \implies x^2 < 9 \implies |x|<3$.
    \item Thus, $R=3$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=3$: $\sum \frac{(-1)^n 3^{2n}}{9^n} = \sum \frac{(-1)^n 9^n}{9^n} = \sum (-1)^n$. Diverges (Divergence Test).
            \item $x=-3$: $\sum \frac{(-1)^n (-3)^{2n}}{9^n} = \sum \frac{(-1)^n 9^n}{9^n} = \sum (-1)^n$. Diverges (Divergence Test).
        \end{itemize}
    \item \textbf{Answer:} $R=3, I=(-3,3)$.
\end{itemize}

\textbf{8. $\displaystyle \sum_{n=1}^{\infty} \frac{x^n}{5n+1}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{x^{n+1}}{5(n+1)+1} \cdot \frac{5n+1}{x^n} \right| = |x| \lim_{n \to \infty} \frac{5n+1}{5n+6} = |x|$.
    \item For convergence, $|x|<1$. Thus, $R=1$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=1$: $\sum \frac{1}{5n+1}$. Diverges (Limit Comparison Test with $\sum \frac{1}{n}$).
            \item $x=-1$: $\sum \frac{(-1)^n}{5n+1}$. Converges (AST).
        \end{itemize}
    \item \textbf{Answer:} $R=1, I=[-1,1)$.
\end{itemize}

\textbf{9. $\displaystyle \sum_{n=1}^{\infty} \frac{(n+1)x^n}{n^2+2n}$}
\begin{itemize}
    \item Simplify term: $\frac{n+1}{n(n+2)}x^n$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(n+2)x^{n+1}}{(n+1)(n+3)} \cdot \frac{n(n+2)}{(n+1)x^n} \right| = |x| \lim_{n \to \infty} \frac{n(n+2)^2}{(n+1)^2(n+3)} = |x| \cdot 1 = |x|$.
    \item For convergence, $|x|<1$. Thus, $R=1$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=1$: $\sum \frac{n+1}{n(n+2)}$. Diverges (Limit Comparison Test with $\sum \frac{1}{n}$).
            \item $x=-1$: $\sum \frac{(-1)^n(n+1)}{n(n+2)}$. Converges (AST).
        \end{itemize}
    \item \textbf{Answer:} $R=1, I=[-1,1)$.
\end{itemize}

\textbf{10. $\displaystyle \sum_{n=1}^{\infty} \frac{(-2)^n x^n}{\sqrt{n}}$}
\begin{itemize}
    \item Term is $\frac{(-1)^n 2^n x^n}{\sqrt{n}}$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(-2)^{n+1} x^{n+1}}{\sqrt{n+1}} \cdot \frac{\sqrt{n}}{(-2)^n x^n} \right| = |-2x| \lim_{n \to \infty} \sqrt{\frac{n}{n+1}} = 2|x|$.
    \item For convergence, $2|x|<1 \implies |x|<1/2$. Thus, $R=1/2$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=1/2$: $\sum \frac{(-2)^n (1/2)^n}{\sqrt{n}} = \sum \frac{(-1)^n}{\sqrt{n}}$. Converges (AST).
            \item $x=-1/2$: $\sum \frac{(-2)^n (-1/2)^n}{\sqrt{n}} = \sum \frac{1^n}{\sqrt{n}} = \sum \frac{1}{n^{1/2}}$. Diverges (p-series, $p=1/2 \le 1$).
        \end{itemize}
    \item \textbf{Answer:} $R=1/2, I=(-1/2, 1/2]$.
\end{itemize}

\subsection*{Solutions to Series Centered at a $\neq$ 0}
\textbf{11. $\displaystyle \sum_{n=1}^{\infty} \frac{(x-5)^n}{n^2}$}
\begin{itemize}
    \item Center $a=5$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(x-5)^{n+1}}{(n+1)^2} \cdot \frac{n^2}{(x-5)^n} \right| = |x-5| \lim_{n \to \infty} \left(\frac{n}{n+1}\right)^2 = |x-5|$.
    \item For convergence, $|x-5|<1$. Thus, $R=1$. Interval: $(4,6)$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=6$: $\sum \frac{1^n}{n^2} = \sum \frac{1}{n^2}$. Converges (p-series, $p=2>1$).
            \item $x=4$: $\sum \frac{(-1)^n}{n^2}$. Converges (absolutely).
        \end{itemize}
    \item \textbf{Answer:} $R=1, I=[4,6]$.
\end{itemize}

\textbf{12. $\displaystyle \sum_{n=1}^{\infty} \frac{(x+1)^n}{n \cdot 4^n}$}
\begin{itemize}
    \item Center $a=-1$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(x+1)^{n+1}}{(n+1)4^{n+1}} \cdot \frac{n \cdot 4^n}{(x+1)^n} \right| = \frac{|x+1|}{4} \lim_{n \to \infty} \frac{n}{n+1} = \frac{|x+1|}{4}$.
    \item For convergence, $\frac{|x+1|}{4}<1 \implies |x+1|<4$. Thus, $R=4$. Interval: $(-5,3)$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=3$: $\sum \frac{4^n}{n \cdot 4^n} = \sum \frac{1}{n}$. Diverges (harmonic).
            \item $x=-5$: $\sum \frac{(-4)^n}{n \cdot 4^n} = \sum \frac{(-1)^n}{n}$. Converges (AST).
        \end{itemize}
    \item \textbf{Answer:} $R=4, I=[-5,3)$.
\end{itemize}

\textbf{13. $\displaystyle \sum_{n=0}^{\infty} \frac{(-1)^n (x-2)^n}{3n+1}$}
\begin{itemize}
    \item Center $a=2$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(x-2)^{n+1}}{3(n+1)+1} \cdot \frac{3n+1}{(x-2)^n} \right| = |x-2| \lim_{n \to \infty} \frac{3n+1}{3n+4} = |x-2|$.
    \item For convergence, $|x-2|<1$. Thus, $R=1$. Interval: $(1,3)$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=3$: $\sum \frac{(-1)^n (1)^n}{3n+1} = \sum \frac{(-1)^n}{3n+1}$. Converges (AST).
            \item $x=1$: $\sum \frac{(-1)^n (-1)^n}{3n+1} = \sum \frac{1}{3n+1}$. Diverges (LCT with $\sum \frac{1}{n}$).
        \end{itemize}
    \item \textbf{Answer:} $R=1, I=(1,3]$.
\end{itemize}

\textbf{14. $\displaystyle \sum_{n=1}^{\infty} \frac{n(x+3)^n}{5^n}$}
\begin{itemize}
    \item Center $a=-3$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(n+1)(x+3)^{n+1}}{5^{n+1}} \cdot \frac{5^n}{n(x+3)^n} \right| = \frac{|x+3|}{5} \lim_{n \to \infty} \frac{n+1}{n} = \frac{|x+3|}{5}$.
    \item For convergence, $|x+3|<5$. Thus, $R=5$. Interval: $(-8,2)$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=2$: $\sum \frac{n \cdot 5^n}{5^n} = \sum n$. Diverges (Divergence Test).
            \item $x=-8$: $\sum \frac{n (-5)^n}{5^n} = \sum (-1)^n n$. Diverges (Divergence Test).
        \end{itemize}
    \item \textbf{Answer:} $R=5, I=(-8,2)$.
\end{itemize}

\textbf{15. $\displaystyle \sum_{n=1}^{\infty} \frac{(3x-1)^n}{n^3}$}
\begin{itemize}
    \item Rewrite: $\frac{[3(x-1/3)]^n}{n^3} = \frac{3^n(x-1/3)^n}{n^3}$. Center $a=1/3$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{3^{n+1}(x-1/3)^{n+1}}{(n+1)^3} \cdot \frac{n^3}{3^n(x-1/3)^n} \right| = 3|x-1/3|$.
    \item For convergence, $3|x-1/3|<1 \implies |x-1/3|<1/3$. Thus, $R=1/3$. Interval: $(0,2/3)$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=2/3$: $\sum \frac{(2-1)^n}{n^3} = \sum \frac{1}{n^3}$. Converges (p-series, $p=3>1$).
            \item $x=0$: $\sum \frac{(-1)^n}{n^3}$. Converges (absolutely).
        \end{itemize}
    \item \textbf{Answer:} $R=1/3, I=[0, 2/3]$.
\end{itemize}

\textbf{16. $\displaystyle \sum_{n=1}^{\infty} \frac{(2x+5)^n}{n^2 4^n}$}
\begin{itemize}
    \item Rewrite: $\frac{[2(x+5/2)]^n}{n^2 4^n} = \frac{2^n(x+5/2)^n}{n^2 4^n} = \frac{(x+5/2)^n}{n^2 2^n}$. Center $a=-5/2$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(x+5/2)^{n+1}}{(n+1)^2 2^{n+1}} \cdot \frac{n^2 2^n}{(x+5/2)^n} \right| = \frac{|x+5/2|}{2}$.
    \item For convergence, $\frac{|x+5/2|}{2}<1 \implies |x+5/2|<2$. Thus, $R=2$. Interval: $(-9/2, -1/2)$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=-1/2$: $\sum \frac{(-1/2+5/2)^n}{n^2 2^n} = \sum \frac{2^n}{n^2 2^n} = \sum \frac{1}{n^2}$. Converges (p-series, $p=2>1$).
            \item $x=-9/2$: $\sum \frac{(-9/2+5/2)^n}{n^2 2^n} = \sum \frac{(-2)^n}{n^2 2^n} = \sum \frac{(-1)^n}{n^2}$. Converges (absolutely).
        \end{itemize}
    \item \textbf{Answer:} $R=2, I=[-9/2, -1/2]$.
\end{itemize}

\subsection*{Solutions to Series with Factorials \& Special Radii}
\textbf{17. $\displaystyle \sum_{n=0}^{\infty} \frac{x^n}{n!}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{x^{n+1}}{(n+1)!} \cdot \frac{n!}{x^n} \right| = \lim_{n \to \infty} \left| \frac{x}{n+1} \right| = |x| \cdot 0 = 0$.
    \item Since $L=0 < 1$ for all $x$, the series converges for all real numbers.
    \item \textbf{Answer:} $R=\infty, I=(-\infty, \infty)$.
\end{itemize}

\textbf{18. $\displaystyle \sum_{n=1}^{\infty} \frac{(2n)!}{(n!)^2} x^n$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(2(n+1))! x^{n+1}}{((n+1)!)^2} \cdot \frac{(n!)^2}{(2n)! x^n} \right| = |x| \lim_{n \to \infty} \frac{(2n+2)!}{(2n)!} \frac{(n!)^2}{((n+1)!)^2}$.
    \item $L = |x| \lim_{n \to \infty} (2n+2)(2n+1) \frac{1}{(n+1)^2} = |x| \lim_{n \to \infty} \frac{4n^2+6n+2}{n^2+2n+1} = 4|x|$.
    \item For convergence, $4|x|<1 \implies |x|<1/4$. Thus, $R=1/4$. Endpoint analysis is complex and often omitted in standard courses, but both endpoints diverge.
    \item \textbf{Answer:} $R=1/4, I=(-1/4, 1/4)$.
\end{itemize}

\textbf{19. $\displaystyle \sum_{n=1}^{\infty} \frac{n! x^n}{100^n}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(n+1)! x^{n+1}}{100^{n+1}} \cdot \frac{100^n}{n! x^n} \right| = \lim_{n \to \infty} \left| \frac{(n+1)x}{100} \right| = \infty$ for any $x \neq 0$.
    \item The series only converges at its center, $x=0$.
    \item \textbf{Answer:} $R=0, I=\{0\}$.
\end{itemize}

\textbf{20. $\displaystyle \sum_{n=0}^{\infty} \frac{(-1)^n x^{2n+1}}{(2n+1)!}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{x^{2(n+1)+1}}{(2(n+1)+1)!} \cdot \frac{(2n+1)!}{x^{2n+1}} \right| = \lim_{n \to \infty} \left| \frac{x^2}{(2n+3)(2n+2)} \right| = |x^2| \cdot 0 = 0$.
    \item Since $L=0 < 1$ for all $x$, the series converges for all real numbers.
    \item \textbf{Answer:} $R=\infty, I=(-\infty, \infty)$.
\end{itemize}

\textbf{21. $\displaystyle \sum_{n=1}^{\infty} \frac{n! (x-4)^n}{n^3 2^n}$}
\begin{itemize}
    \item Center $a=4$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(n+1)! (x-4)^{n+1}}{(n+1)^3 2^{n+1}} \cdot \frac{n^3 2^n}{n! (x-4)^n} \right| = \lim_{n \to \infty} \left| \frac{(n+1)(x-4)}{2} \left(\frac{n}{n+1}\right)^3 \right| = \infty$ for $x \neq 4$.
    \item The series only converges at its center, $x=4$.
    \item \textbf{Answer:} $R=0, I=\{4\}$.
\end{itemize}

\subsection*{Solutions to Series for the Root Test}
\textbf{22. $\displaystyle \sum_{n=1}^{\infty} \left(\frac{n}{3n+1}\right)^n x^n$}
\begin{itemize}
    \item \textbf{Root Test:} $L = \lim_{n \to \infty} \sqrt[n]{\left| \left(\frac{n}{3n+1}\right)^n x^n \right|} = \lim_{n \to \infty} \left|\frac{n x}{3n+1}\right| = |x| \lim_{n \to \infty} \frac{n}{3n+1} = \frac{|x|}{3}$.
    \item For convergence, $\frac{|x|}{3}<1 \implies |x|<3$. Thus, $R=3$. Endpoints diverge by Divergence Test.
    \item \textbf{Answer:} $R=3, I=(-3,3)$.
\end{itemize}

\textbf{23. $\displaystyle \sum_{n=1}^{\infty} \left(1+\frac{1}{n}\right)^{n^2} x^n$}
\begin{itemize}
    \item \textbf{Root Test:} $L = \lim_{n \to \infty} \sqrt[n]{\left| \left(1+\frac{1}{n}\right)^{n^2} x^n \right|} = |x| \lim_{n \to \infty} \left(1+\frac{1}{n}\right)^n = |x| \cdot e = e|x|$.
    \item For convergence, $e|x|<1 \implies |x|<1/e$. Thus, $R=1/e$. Endpoints diverge.
    \item \textbf{Answer:} $R=1/e, I=(-1/e, 1/e)$.
\end{itemize}

\textbf{24. $\displaystyle \sum_{n=2}^{\infty} \frac{(x+2)^n}{(\ln n)^n}$}
\begin{itemize}
    \item Center $a=-2$.
    \item \textbf{Root Test:} $L = \lim_{n \to \infty} \sqrt[n]{\left| \frac{(x+2)^n}{(\ln n)^n} \right|} = \lim_{n \to \infty} \left|\frac{x+2}{\ln n}\right| = |x+2| \cdot 0 = 0$.
    \item Since $L=0<1$ for all $x$, the series converges everywhere.
    \item \textbf{Answer:} $R=\infty, I=(-\infty, \infty)$.
\end{itemize}

\textbf{25. $\displaystyle \sum_{n=1}^{\infty} \left(\frac{4n+3}{2n+5}\right)^n (x-1)^n$}
\begin{itemize}
    \item Center $a=1$.
    \item \textbf{Root Test:} $L = \lim_{n \to \infty} \sqrt[n]{\left| \left(\frac{4n+3}{2n+5}\right)^n (x-1)^n \right|} = |x-1| \lim_{n \to \infty} \frac{4n+3}{2n+5} = 2|x-1|$.
    \item For convergence, $2|x-1|<1 \implies |x-1|<1/2$. Thus, $R=1/2$. Endpoints diverge by Divergence Test.
    \item \textbf{Answer:} $R=1/2, I=(1/2, 3/2)$.
\end{itemize}

\subsection*{Solutions to Term-by-Term Differentiation and Integration}
\textbf{26. Find a series for $\ln(1+x)$.}
\begin{itemize}
    \item $\ln(1+x) = \int \frac{1}{1+x} dx = \int \left( \sum_{n=0}^{\infty} (-1)^n x^n \right) dx = C + \sum_{n=0}^{\infty} (-1)^n \frac{x^{n+1}}{n+1}$.
    \item To find C, let $x=0$. $\ln(1)=0$, and the series is also 0. So $C=0$.
    \item Series is $\sum_{n=0}^{\infty} \frac{(-1)^n x^{n+1}}{n+1} = x - \frac{x^2}{2} + \frac{x^3}{3} - \dots$
    \item The radius of convergence is the same, $R=1$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=1$: $\sum \frac{(-1)^n}{n+1}$. Converges (AST).
            \item $x=-1$: $\sum \frac{(-1)^n (-1)^{n+1}}{n+1} = \sum \frac{-1}{n+1}$. Diverges (harmonic series).
        \end{itemize}
    \item \textbf{Answer:} Series: $\displaystyle \sum_{n=1}^{\infty} \frac{(-1)^{n-1} x^n}{n}$, $R=1$, $I=(-1,1]$.
\end{itemize}

\textbf{27. Find a series for $\frac{1}{(1+x)^2}$.}
\begin{itemize}
    \item Note that $\frac{d}{dx}\left(\frac{1}{1+x}\right) = -\frac{1}{(1+x)^2}$.
    \item $\frac{d}{dx} \left( \sum_{n=0}^{\infty} (-1)^n x^n \right) = \sum_{n=1}^{\infty} (-1)^n n x^{n-1}$.
    \item So, $\frac{1}{(1+x)^2} = - \sum_{n=1}^{\infty} (-1)^n n x^{n-1} = \sum_{n=1}^{\infty} (-1)^{n+1} n x^{n-1} = 1 - 2x + 3x^2 - \dots$
    \item The radius is the same, $R=1$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=1$: $\sum (-1)^{n+1} n$. Diverges (Divergence Test).
            \item $x=-1$: $\sum (-1)^{n+1} n (-1)^{n-1} = \sum n$. Diverges (Divergence Test).
        \end{itemize}
    \item \textbf{Answer:} Series: $\displaystyle \sum_{n=0}^{\infty} (-1)^n (n+1) x^n$, $R=1$, $I=(-1,1)$.
\end{itemize}

\subsection*{Solutions to Mixed and Challenging Problems}
\textbf{28. $\displaystyle \sum_{n=1}^{\infty} \frac{3^n (x-1)^{2n}}{n}$}
\begin{itemize}
    \item Let $u = 3(x-1)^2$. The series becomes $\sum \frac{u^n}{n \cdot 3^n}$ without the $3^n$ factor, or let $u = (x-1)^2$ giving $\sum \frac{3^n u^n}{n}$. Let's use the Ratio Test directly.
    \item $L = \lim_{n \to \infty} \left| \frac{3^{n+1} (x-1)^{2(n+1)}}{n+1} \cdot \frac{n}{3^n (x-1)^{2n}} \right| = \lim_{n \to \infty} \left| 3(x-1)^2 \frac{n}{n+1} \right| = 3(x-1)^2$.
    \item For convergence, $3(x-1)^2 < 1 \implies (x-1)^2 < 1/3 \implies |x-1| < 1/\sqrt{3}$.
    \item $R=1/\sqrt{3}$. Interval: $(1-1/\sqrt{3}, 1+1/\sqrt{3})$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $|x-1|=1/\sqrt{3} \implies (x-1)^2 = 1/3$. The series becomes $\sum \frac{3^n(1/3)^n}{n} = \sum \frac{1}{n}$. Diverges. At both endpoints, we get divergence. Wait, let me recheck.
            \item Correct endpoint test: At $x = 1 \pm 1/\sqrt{3}$, we have $(x-1)^2=1/3$. The series becomes $\sum \frac{3^n (1/3)^n}{n} = \sum \frac{1}{n}$. This diverges.
        \end{itemize}
    \item \textbf{Answer:} $R=1/\sqrt{3}, I=(1-1/\sqrt{3}, 1+1/\sqrt{3})$.
\end{itemize}

\textbf{29. $\displaystyle \sum_{n=1}^{\infty} \frac{n^2 (x+2)^n}{2^n (n^3+1)}$}
\begin{itemize}
    \item Center $a=-2$.
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{(n+1)^2(x+2)^{n+1}}{2^{n+1}((n+1)^3+1)} \cdot \frac{2^n(n^3+1)}{n^2(x+2)^n} \right| = \frac{|x+2|}{2} \lim_{n \to \infty} \frac{(n+1)^2(n^3+1)}{n^2((n+1)^3+1)} = \frac{|x+2|}{2}$.
    \item For convergence, $|x+2|<2$. Thus, $R=2$. Interval: $(-4,0)$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=0$: $\sum \frac{n^2 2^n}{2^n(n^3+1)} = \sum \frac{n^2}{n^3+1}$. Diverges (LCT with $\sum \frac{1}{n}$).
            \item $x=-4$: $\sum \frac{n^2 (-2)^n}{2^n(n^3+1)} = \sum \frac{(-1)^n n^2}{n^3+1}$. Converges (AST).
        \end{itemize}
    \item \textbf{Answer:} $R=2, I=[-4,0)$.
\end{itemize}

\textbf{30. $\displaystyle \sum_{n=1}^{\infty} \frac{(n!)^3}{(3n)!} x^n$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{((n+1)!)^3 x^{n+1}}{(3(n+1))!} \cdot \frac{(3n)!}{(n!)^3 x^n} \right| = |x| \lim_{n \to \infty} \frac{(n+1)^3 (n!)^3}{(n!)^3} \frac{(3n)!}{(3n+3)!}$.
    \item $L = |x| \lim_{n \to \infty} (n+1)^3 \frac{1}{(3n+3)(3n+2)(3n+1)} = |x| \lim_{n \to \infty} \frac{n^3+\dots}{27n^3+\dots} = \frac{|x|}{27}$.
    \item For convergence, $|x|<27$. Thus, $R=27$. Endpoints diverge.
    \item \textbf{Answer:} $R=27, I=(-27,27)$.
\end{itemize}

\textbf{31. $\displaystyle \sum_{n=1}^{\infty} \frac{(-1)^n (x-4)^n}{\sqrt{n^2+9}}$}
\begin{itemize}
    \item Center $a=4$.
    \item \textbf{Ratio Test:} $L = |x-4| \lim_{n \to \infty} \frac{\sqrt{n^2+9}}{\sqrt{(n+1)^2+9}} = |x-4| \cdot 1 = |x-4|$.
    \item For convergence, $|x-4|<1$. Thus, $R=1$. Interval: $(3,5)$.
    \item \textbf{Endpoints:}
        \begin{itemize}
            \item $x=5$: $\sum \frac{(-1)^n}{\sqrt{n^2+9}}$. Converges (AST).
            \item $x=3$: $\sum \frac{(-1)^n (-1)^n}{\sqrt{n^2+9}} = \sum \frac{1}{\sqrt{n^2+9}}$. Diverges (LCT with $\sum \frac{1}{n}$).
        \end{itemize}
    \item \textbf{Answer:} $R=1, I=(3,5]$.
\end{itemize}

\textbf{32. $\displaystyle \sum_{n=1}^{\infty} \frac{x^{3n}}{n!}$}
\begin{itemize}
    \item \textbf{Ratio Test:} $L = \lim_{n \to \infty} \left| \frac{x^{3(n+1)}}{(n+1)!} \cdot \frac{n!}{x^{3n}} \right| = \lim_{n \to \infty} \left| \frac{x^3}{n+1} \right| = |x^3| \cdot 0 = 0$.
    \item Since $L=0<1$ for all $x$, the series converges everywhere.
    \item \textbf{Answer:} $R=\infty, I=(-\infty, \infty)$.
\end{itemize}
